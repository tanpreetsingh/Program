public class Main
{
    public static void main(String[] args)
    {
                        int rows = 3;
			for (int i = 1; i <= rows; i++) 
	                { 
	                        for (int j = rows; j > i; j--)
				{
					System.out.print("  ");
				}
				for (int k = 1; k <= (i * 4) - 3; k++) 
	                        { 
	                                System.out.print("*"); 
	                        } 
	                        System.out.println(); 
	                } 
	                for (int i = rows - 1; i >= 1; i--)
			{
				for (int j = rows - 1; j >= i; j--)
				{
					System.out.print("  ");
				}
				for (int k = 1; k <= (i * 4) - 3; k++)
				{
					System.out.print("*");
				}
				System.out.println();
			}
		}
	}
